static void expon_reinit(sk_expon *e)
{
    SKFLT onedsr;
    e->stime = 0;
    e->sdur = e->dur * e->sr;
    onedsr = 1.0 / e->sr;

    if ((e->a * e->b) > 0.0) {
        e->incr = pow((SKFLT)(e->b / e->a), onedsr / e->dur);
    } else {
        e->incr = 1;
        e->val = e->a;
    }

    e->val = e->a;
}
#+END_SRC

#+NAME: funcs
#+BEGIN_SRC c
SKFLT sk_expon_tick(sk_expon *e, SKFLT trig)
{
    SKFLT out;

    out = 0;

    if (trig != 0) {
        expon_reinit(e);
        e->init = 0;
    }

    if (e->init) {
        out = 0;
    }

    if (e->stime < e->sdur) {
        SKFLT val = e->val;
        e->val *= e->incr;
        e->stime++;
        out = val;
    } else {
        out = e->b;
    }

    return out;
}
